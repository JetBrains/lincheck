// Test cases for await-loop detection in control-flow graph analysis.
// An await loop is a loop that:
//   - contains at least one shared read (field/array read)
//   - contains NO shared writes, monitor operations, or side-effecting calls
//     (except Thread.onSpinWait)

public class AwaitLoopCases {

    // Shared fields used in test cases
    volatile boolean flag;
    volatile int counter;
    int[] array = new int[10];

    // --- Case 1: Pure await loop (only field reads) ---
    // Should be detected as an await loop.
    public void awaitFieldRead() {
        while (!flag) {
            // busy-wait: only reads a volatile field
        }
    }

    // --- Case 2: Non-await loop with a field write ---
    // Should NOT be detected as an await loop.
    public void nonAwaitFieldWrite() {
        while (!flag) {
            counter = 42;
        }
    }

    // --- Case 3: Non-await loop with a method call ---
    // Should NOT be detected as an await loop (method call is a side-effect).
    public void nonAwaitWithMethodCall() {
        while (!flag) {
            doSomething();
        }
    }

    private void doSomething() {
        // some side-effecting call
    }

    // --- Case 4: Non-await loop with monitor enter ---
    // Should NOT be detected as an await loop.
    public void nonAwaitWithMonitor() {
        while (!flag) {
            synchronized (this) {
                // monitor enter/exit
            }
        }
    }

    // --- Case 5: Await loop with array read only ---
    // Should be detected as an await loop.
    public int awaitArrayRead() {
        int sum = 0;
        while (sum < 100) {
            sum = array[0];
        }
        return sum;
    }

    // --- Case 6: Non-await loop with array write ---
    // Should NOT be detected as an await loop.
    public void nonAwaitArrayWrite() {
        while (!flag) {
            array[0] = 1;
        }
    }

    // --- Case 7: No loops at all (straight-line code) ---
    // No loops, so no await loops either.
    public int noLoop() {
        return counter + 1;
    }

    // --- Case 8: Multiple loops, one await and one not ---
    // First loop is await (only reads), second is not (has a write).
    public void mixedLoops() {
        while (!flag) {
            // await: only reads
        }
        while (!flag) {
            counter = 0; // write => not await
        }
    }

    // --- Case 9: Await loop with both field and array reads ---
    // Should be detected as an await loop.
    public void awaitMixedReads() {
        while (!flag) {
            int x = counter;
            int y = array[0];
        }
    }

    // --- Case 10: Non-await loop — reads AND writes (counter++) ---
    // This loop reads and writes, so it's not await.
    public void nonAwaitReadAndWrite() {
        while (counter < 10) {
            counter++;
        }
    }

    // === Path-sensitive test cases ===
    // These test loops where SOME paths through the loop body are read-only
    // (spin-retry paths) while other paths have writes or calls.
    // The new algorithm should detect the read-only back-edge path.

    // --- Case 11: Loop with CAS on one path, read-only spin on another ---
    // Simulates: check element, if null → spin (return@loop), else CAS + return
    // Should be detected as an await loop (the null-check spin path is read-only).
    volatile int head;
    volatile int tail;
    volatile int state;
    java.util.concurrent.atomic.AtomicInteger atomicState = new java.util.concurrent.atomic.AtomicInteger();

    public int awaitLoopWithCasOnAlternatePath() {
        while (true) {
            int element = array[head];
            if (element == 0) {
                // spin-retry: read-only path back to loop header
                continue;
            }
            // non-spinning path: CAS + write (side effects)
            if (atomicState.compareAndSet(0, 1)) {
                array[head] = 0;
                return element;
            }
        }
    }

    // --- Case 12: Loop with write on one branch, read-only spin on another ---
    // Simulates: read flag, if condition A → spin, if condition B → write + exit
    // Should be detected as an await loop.
    public void awaitLoopWithWriteOnAlternatePath() {
        while (true) {
            int s = state;
            if (s == 0) {
                // spin-retry path: no writes, just continue
                continue;
            }
            // exit path with a write
            counter = s;
            return;
        }
    }

    // --- Case 13: Loop where ALL paths have side effects ---
    // No read-only spin path exists.
    // Should NOT be detected as an await loop.
    public void nonAwaitAllPathsDirty() {
        while (true) {
            int s = state;
            if (s == 0) {
                counter = 42; // write on this path too
                continue;
            }
            counter = s;
            return;
        }
    }

    // --- Case 14: Loop with method call on spin path ---
    // The spin path has a method call (side effect).
    // Should NOT be detected as an await loop.
    public void nonAwaitCallOnSpinPath() {
        while (true) {
            int s = state;
            if (s == 0) {
                doSomething(); // method call on the spin path
                continue;
            }
            return;
        }
    }

    // --- Case 15: Loop with only reads and Thread.onSpinWait on spin path ---
    // Thread.onSpinWait is explicitly allowed.
    // Should be detected as an await loop.
    public void awaitLoopWithOnSpinWait() {
        while (true) {
            int s = state;
            if (s != 0) {
                return;
            }
            Thread.onSpinWait();
        }
    }

    // --- Case 16: Multi-path loop where one spin path has reads, another spin path has writes ---
    // Two back-edge paths: one is clean (read-only), one is dirty (has writes).
    // Should be detected as await loop because at least one clean path exists.
    public void awaitLoopOneCleanOneNot() {
        while (true) {
            int s = state;
            if (s == 0) {
                // clean spin path
                continue;
            }
            if (s == 1) {
                counter = s; // dirty spin path
                continue;
            }
            return;
        }
    }
}
